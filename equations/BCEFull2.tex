    \setlength{\jot}{\baselineskip}
    \begin{split}
        \log(1-\sigma(x)) &= \log\left(1-\frac{e^x}{e^x+1}\right)
        \\ &= \log\left(\frac{e^x+1-e^x}{e^x+1}\right)
        \\ &= \log\left(\frac{1}{e^x+1}\right)
        \\ &= \log(1) - \log(e^x+1) \\ &= -\log(1+e^x)
    \end{split}
\end{equation*}
